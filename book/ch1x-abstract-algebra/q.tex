\begin{chapex}
For each of the following binary operations on $\mathbb{Z}$, determine whether it is: (i) associative, (ii) unital, (iii) commutative.
\begin{multicols}{2}
\begin{enumerate}[(a)]
\item $a \star b = ab$
\item $a \star b = ab + a + b$
\item $a \star b = a^b$
\item $a \star b = |b - a|$
\item $a \star b = a$
\item $a \star b = a + b + 1$
\end{enumerate}
\end{multicols}
\end{chapex}
